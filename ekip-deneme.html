        function updateDashboard() {

            const hotelKey = document.getElementById('hotelSelector').value;

            const data = allData[hotelKey];


            // Update Stats

            document.getElementById('statClicks').innerText = data.clicks.reduce((a, b) => a + b, 0).toLocaleString();

            const tbody = document.getElementById('dataTableBody');

            tbody.innerHTML = '';

            data.labels.forEach((label, i) => {

                tbody.innerHTML += `

                    <tr class="hover:bg-slate-50 transition-colors">

                    </tr>

                `;
